Deleting all empties str inside the value (suppression des vides dans une chaine)

In[]:
#Cette fonction supprime tous les points, les espaces et vigules dans un montant (car nos valeurs sont entires)
def str_with_point_or_comma(v):
    try:
        k = str(v).strip()

        if ' ' in k:
            l = k.split(' ')
            
            # On supprime les valeurs vide de la liste
            for i in range(len(l)-1, -1, -1):
                if l[i] == "":
                    del l[i]
            
            if len(l)==1:
                return int(l[0])
            else:
                concatenation = ''.join(str(element) for element in l) # On fait la concatnation des entiers converti en str
                
                if '.' in concatenation:
                    l = concatenation.split('.')
                    concatenation = ''.join(str(element) for element in l)

                    if ',' in concatenation:
                        l = concatenation.split(',')
                        concatenation = ''.join(str(element) for element in l)
                        return int(concatenation)
                    return int(concatenation)
                return int(concatenation)
            
        elif '.' in k:
            l = k.split('.')
            concatenation = ''.join(str(element) for element in l)
            return int(concatenation)
        
        elif ',' in k:
            l = k.split(',')
            concatenation = ''.join(str(element) for element in l)
            return int(concatenation)
        
        return int(k)
    except:
        return v
